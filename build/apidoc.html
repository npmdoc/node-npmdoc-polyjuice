function polyjuice.from.eslint (filename, log)

  • description and source-code
    from.eslint = function (filename, log) {
  var cli = new CLIEngine()
  var data = {}

  filename.forEach(function(file) {
    var read = cli.getConfigForFile(file)
    data = objectAssign(data, read.rules)
  })

  var rulesPath = path.join(__dirname, '../dictionaries/eslint/')

  var dictionary = {}

  fs.readdirSync(rulesPath).forEach(function(name) {
    var rule = path.join(rulesPath, name)

    dictionary[name.split('.')[0]] = require(rule)
  })

  return interpreter(dictionary, data, truth_tbl, log)
}

function polyjuice.from.jshint (filename, log, encoding)

  • description and source-code
    from.jshint = function (filename, log, encoding) {
  encoding = encoding || 'utf8'

  var data = {}

  filename.forEach(function(file) {
    var read = JSON.parse(stripJsonComments(fs.readFileSync(file, encoding)))
    data = objectAssign(data, read)
  })

  var output = {}

  var rulesPath = path.join(__dirname, '../dictionaries/jshint/')

  var dictionary = {}

  fs.readdirSync(rulesPath).forEach(function(name) {
    var rule = path.join(rulesPath, name)

    dictionary[name.split('.')[0]] = require(rule)
  });

  output.env = {}

  dictionary._env.forEach(function(env) {
    var from = env
    var to = env

    if (typeof env !== 'string') {
      from = env[0]
      to = env[1]
    }

    if (from in data && data[from]) {
      output.env[to] = true
    }
  })

  output.globals = data.globals || {}

  output.rules = interpreter(dictionary, data, truth_tbl, log)

  return output
}

function polyjuice.to.eslint (jshint_src, jscs_src, log)

  • description and source-code
    to.eslint = function (jshint_src, jscs_src, log) {
  var jshint_json = jshint(jshint_src, log)
  var jscs_json = jscs(jscs_src, log)

  for (var name in jscs_json.rules) {
    jshint_json.rules[name] = jscs_json.rules[name]
  }

  return jshint_json
}
